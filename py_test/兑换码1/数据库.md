### main.exchange_code 兑换码(增加)

```sql
CREATE TABLE `exchange_code`
(
    `id`               bigint       NOT NULL COMMENT '自增id',
    `name`             varchar(200) NOT NULL DEFAULT '' COMMENT '活动名称',
    `reward`           json         NOT NULL COMMENT '奖励内容物品列表',
    `code`             varchar(64)  NOT NULL COMMENT '兑换码',
    `count`            int          NOT NULL DEFAULT 0 COMMENT '兑换数量',
    `exchanged_count`  int          NOT NULL DEFAULT 0 COMMENT '已兑换数量',
    `disabled_time`    bigint       NOT NULL DEFAULT 0 COMMENT '禁用时间毫秒戳:  0.正常  大于0时为失效的时间  -1为永久有效',
    `status`           tinyint      NOT NULL DEFAULT 0 COMMENT '0: 未启用 1: 已启用 2: 已结束',
    `create_time`      bigint       NOT NULL DEFAULT 0 COMMENT '创建时间',
    PRIMARY KEY (`id`),
    UNIQUE `uk_code` (`code`),
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='兑换码表';
```

### main.user_exchange_code 用户兑换码(增加)

```sql
CREATE TABLE `user_exchange_code`
(
    `id`               bigint       NOT NULL COMMENT '自增id',
    `code`             varchar(64)  NOT NULL COMMENT '兑换码',
    `user_id`          bigint       NOT NULL COMMENT '用户ID',
    `exchange_time`    bigint       NOT NULL COMMENT '兑换时间',
    PRIMARY KEY (`id`),
    UNIQUE `uk_user_code` (`code`, `user_id`),
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户兑换码表';
```
